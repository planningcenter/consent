# Pagination

> A navigation element for pagination results list and table results

export function Pagination({ pages }) {
  // TODO: rename prev/next from Page and something incnlusive of first/next
  return (
    <Container>
      <Page href="#previous">Previous</Page>
      <Page href="#next">Next</Page>
      {[
        ...Array(pages).map((page) => (
          <Page href={`${rootURL}/${page}`}>page</Page>
        )),
      ]}
      <TruncatedPages />
    </Container>
  );
}

export function Container(props) {
  // TODO: make aria-label a customizable default
  return (
    <nav aria-label="Page navigation example">
      <ul data-pagination {...props} />
    </nav>
  );
}

export function TruncatedPages(props) {
  return (
    <li data-pagination-item>
      <span {...props} data-visually-hidden>
        â€¦
      </span>
    </li>
  );
}

export function Page(props) {
  return (
    <li data-pagination-item>
      <a data-pagination-link {...props} />
    </li>
  );
}
